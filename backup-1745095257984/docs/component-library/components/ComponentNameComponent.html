import { Component, Input, Output, EventEmitter, ChangeDetectionStrategy } from '@angular/core';
import { CommonModule } from '@angular/common';

/**
 * Component Template
 *
 * This is a template for creating new components following the DateNight.io
 * component standards. Use this as a starting point for new components.
 *
 * Features:
 * - Standalone component
 * - OnPush change detection
 * - BEM-compliant CSS class naming
 * - Proper input/output naming
 * - Comprehensive documentation
 */
@Component({
  selector: 'app-component-name',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './component-name.component.html',
  styleUrls: ['./component-name.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class ComponentNameComponent {
  /**
   * Primary input property description.
   * Explain what this property does and how it affects the component.
   */
  @Input() inputProperty = '';

  /**
   * Optional input with default value.
   * @default 'default-value'
   */
  @Input() optionalInput = 'default-value';

  /**
   * Boolean flag to control component behavior.
   * Explain what happens when this is true vs false.
   * @default false
   */
  @Input() isEnabled = false;

  /**
   * Event emitted when the primary action occurs.
   * Explain when this event is emitted and what data it contains.
   */
  @Output() actionCompleted = new EventEmitter<string>();

  /**
   * Event emitted when a secondary action occurs.
   * Explain when this event is emitted and what data it contains.
   */
  @Output() secondaryAction = new EventEmitter<void>();

  /**
   * Internal state property.
   * Use private properties for internal component state.
   */
  private _internalState = '';

  /**
   * Getter for internal state.
   * Use getters/setters for properties that need processing.
   */
  get internalState(): string {
    return this._internalState;
  }

  /**
   * Handles the primary action.
   * Explain what this method does and when it should be called.
   */
  handlePrimaryAction(): void {
    // Implementation
    this.actionCompleted.emit(this.inputProperty);
  }

  /**
   * Handles the secondary action.
   * Explain what this method does and when it should be called.
   */
  handleSecondaryAction(): void {
    // Implementation
    this.secondaryAction.emit();
  }

  /**
   * Helper method for internal component logic.
   * Document private methods that contain significant logic.
   * @param value The value to process
   * @returns The processed result
   */
  private processValue(value: string): string {
    // Implementation
    return value.toUpperCase();
  }
}

  Component Template HTML
  
  This is a template for creating new components following the DateNight.io
  component standards. Use this as a starting point for new components.
  
  Guidelines:
  - Use BEM naming convention for CSS classes
  - Keep templates clean and readable
  - Use semantic HTML elements
  - Include appropriate ARIA attributes for accessibility
  - Use ng-container for structural directives when possible
-->

<div class="component-name" [class.component-name--enabled]="isEnabled">
  <!-- Component Header -->
  <div class="component-name__header">
    <h2 class="component-name__title">{{ inputProperty }}</h2>

    @if (optionalInput) {
      <div class="component-name__subtitle">{{ optionalInput }}</div>
    }
  </div>

  <!-- Component Content -->
  <div class="component-name__content">
    <!-- Primary Content Area -->
    <div class="component-name__primary-content">
      <ng-content></ng-content>
    </div>

    <!-- Optional Content Area -->
    @if (isEnabled) {
      <div class="component-name__secondary-content">
        <!-- Secondary content that only appears when enabled -->
        <p class="component-name__description">Additional content when enabled</p>
      </div>
    }
  </div>

  <!-- Component Actions -->
  <div class="component-name__actions">
    <button
      type="button"
      class="component-name__button component-name__button--primary"
      (click)="handlePrimaryAction()"
      [attr.aria-label]="'Perform primary action for ' + inputProperty"
    >
      Primary Action
    </button>

    <button
      type="button"
      class="component-name__button component-name__button--secondary"
      (click)="handleSecondaryAction()"
    >
      Secondary Action
    </button>
  </div>
</div>

/**
 * Component Template SCSS
 * 
 * This is a template for creating new components following the DateNight.io
 * component standards. Use this as a starting point for new components.
 * 
 * Guidelines:
 * - Use BEM naming convention for CSS classes
 * - Import design tokens using @use
 * - Use mixins for typography and common patterns
 * - Keep specificity low
 * - Organize styles in a logical order
 */

// Import design system
@use 'src/app/core/design/main' as ds;

// Component Block
.component-name {
  display: block;
  border-radius: ds.$border-radius-lg;
  background-color: ds.$color-white;
  box-shadow: ds.$shadow-md;
  overflow: hidden;
  transition: all ds.$transition-normal ds.$transition-timing-default;

  // Block modifier
  &--enabled {
    border-left: 4px solid ds.$color-primary;
  }

  // Header element
  &__header {
    padding: ds.$spacing-4;
    border-bottom: 1px solid ds.$color-light-gray-2;
  }

  // Title element
  &__title {
    @include ds.heading-4;
    margin-bottom: ds.$spacing-2;
  }

  // Subtitle element
  &__subtitle {
    @include ds.body-small;
    color: ds.$color-dark-gray-1;
  }

  // Content element
  &__content {
    padding: ds.$spacing-4;
  }

  // Primary content element
  &__primary-content {
    margin-bottom: ds.$spacing-4;
  }

  // Secondary content element
  &__secondary-content {
    padding: ds.$spacing-3;
    background-color: ds.$color-light-gray-1;
    border-radius: ds.$border-radius-md;
    margin-top: ds.$spacing-3;
  }

  // Description element
  &__description {
    @include ds.body-small;
    margin-bottom: 0;
  }

  // Actions element
  &__actions {
    display: flex;
    gap: ds.$spacing-3;
    padding: ds.$spacing-4;
    border-top: 1px solid ds.$color-light-gray-2;
  }

  // Button element
  &__button {
    @include ds.button-text;
    padding: ds.$spacing-2 ds.$spacing-4;
    border-radius: ds.$border-radius-md;
    border: none;
    cursor: pointer;
    transition: all ds.$transition-fast ds.$transition-timing-default;

    // Primary button modifier
    &--primary {
      background-color: ds.$color-primary;
      color: ds.$color-white;

      &:hover {
        background-color: ds.$color-primary-dark;
      }

      &:focus-visible {
        outline: 2px solid rgba(ds.$color-primary, 0.5);
        outline-offset: 2px;
      }

      &:active {
        transform: translateY(1px);
      }
    }

    // Secondary button modifier
    &--secondary {
      background-color: transparent;
      color: ds.$color-primary;
      border: 1px solid ds.$color-primary;

      &:hover {
        background-color: rgba(ds.$color-primary, 0.05);
      }

      &:focus-visible {
        outline: 2px solid rgba(ds.$color-primary, 0.5);
        outline-offset: 2px;
      }

      &:active {
        transform: translateY(1px);
      }
    }
  }
}
