import { Component, Input, ChangeDetectionStrategy } from '@angular/core';
import { CommonModule } from '@angular/common';

/**
 * Card Component
 *
 * A versatile card component that follows the DateNight.io design system.
 * Can be used for displaying content in a contained, styled box.
 */
@Component({
  selector: 'app-card',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './card.component.html',
  styleUrls: ['./card.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class CardComponent {
  /**
   * Whether to show a border around the card.
   * @default true
   */
  @Input() bordered = true;

  /**
   * Whether to apply a shadow to the card.
   * @default true
   */
  @Input() shadowed = true;

  /**
   * The padding size for the card.
   * - 'none': No padding
   * - 'small': Small padding (8px)
   * - 'medium': Medium padding (16px)
   * - 'large': Large padding (24px)
   * @default 'medium'
   */
  @Input() padding: 'none' | 'small' | 'medium' | 'large' = 'medium';

  /**
   * The border radius size for the card.
   * - 'none': No border radius
   * - 'small': Small border radius (4px)
   * - 'medium': Medium border radius (8px)
   * - 'large': Large border radius (12px)
   * @default 'medium'
   */
  @Input() radius: 'none' | 'small' | 'medium' | 'large' = 'medium';

  /**
   * Whether the card is clickable.
   * Adds hover and active states.
   * @default false
   */
  @Input() clickable = false;

  /**
   * Whether the card is in a selected state.
   * @default false
   */
  @Input() selected = false;

  /**
   * Whether the card is in a disabled state.
   * @default false
   */
  @Input() disabled = false;

  /**
   * Gets the CSS classes for the card based on its properties.
   * @returns An object with CSS class names as keys and boolean values
   */
  get cardClasses(): Record<string, boolean> {
    return {
      card: true,
      'card--bordered': this.bordered,
      'card--shadowed': this.shadowed,
      [`card--padding-${this.padding}`]: true,
      [`card--radius-${this.radius}`]: true,
      'card--clickable': this.clickable,
      'card--selected': this.selected,
      'card--disabled': this.disabled,
    };
  }
}

<div [ngClass]="cardClasses">
  <ng-content></ng-content>
</div>

/**
 * Card Component Styles
 * 
 * Implements the DateNight.io card styles using the design system.
 */

@use 'src/app/core/design/main' as ds;

.card {
  display: block;
  background-color: ds.$color-white;
  transition: all ds.$transition-normal ds.$transition-timing-default;

  // Border variants
  &--bordered {
    border: 1px solid ds.$color-light-gray-2;
  }

  // Shadow variants
  &--shadowed {
    box-shadow: ds.$shadow-md;
  }

  // Padding variants
  &--padding-none {
    padding: 0;
  }

  &--padding-small {
    padding: ds.$spacing-2;
  }

  &--padding-medium {
    padding: ds.$spacing-4;
  }

  &--padding-large {
    padding: ds.$spacing-6;
  }

  // Border radius variants
  &--radius-none {
    border-radius: 0;
  }

  &--radius-small {
    border-radius: ds.$border-radius-sm;
  }

  &--radius-medium {
    border-radius: ds.$border-radius-md;
  }

  &--radius-large {
    border-radius: ds.$border-radius-lg;
  }

  // State variants
  &--clickable {
    cursor: pointer;

    &:hover:not(.card--disabled) {
      transform: translateY(-2px);
      box-shadow: ds.$shadow-lg;
    }

    &:active:not(.card--disabled) {
      transform: translateY(0);
      box-shadow: ds.$shadow-md;
    }
  }

  &--selected {
    border-color: ds.$color-primary;
    box-shadow:
      0 0 0 1px ds.$color-primary,
      ds.$shadow-md;
  }

  &--disabled {
    opacity: 0.6;
    cursor: not-allowed;
    pointer-events: none;
  }
}

// Dark mode styles
@media (prefers-color-scheme: dark) {
  body.dark-mode-enabled {
    .card {
      background-color: ds.$color-dark-gray-2;

      &--bordered {
        border-color: ds.$color-dark-gray-1;
      }

      &--selected {
        border-color: ds.$color-primary-light;
        box-shadow:
          0 0 0 1px ds.$color-primary-light,
          ds.$shadow-md;
      }
    }
  }
}
